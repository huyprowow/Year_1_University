function convertToRoman(num) {
  let kq=''
  while(num>0){
    let str= kt(num)
    let loop=Math.floor(num/kt2(num))
    let du=num%kt2(num)
    for(let i=0;i<loop;i++){
      kq+=str
    }
    num=du
    
  }
function kt(a){
  if(a>=1000){
    return 'M'
    }
  if(a<1000&&a>=900){
    return 'CM'
    }
    if(a<900&&a>=500){
    return 'D'
    }
    if(a<500&&a>=400){
    return 'CD'
    }
    if(a<400&&a>=100){
    return 'C'
    }
    if(a<100&&a>=90){
    return 'XC'

    if(a<90&&a>=50){

    return 'XL'

    return 'X'

}
